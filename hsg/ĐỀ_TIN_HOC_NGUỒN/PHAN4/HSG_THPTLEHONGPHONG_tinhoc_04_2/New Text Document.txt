var n,m,i,j,t,s:longint;
    a:array [1..16000] of longint;
    f,f1:text;
begin
assign(f,'c:\maygame.inp');
reset(f);
assign(f1,'c:\maygame.out');
rewrite(f1);
read(f,n,m);
for i:=1 to n do
read(f,a[i]);
for i:=1 to n-1 do
for j:=i+1 to n do
if a[i]>a[j] then
begin
t:=a[i];
a[i]:=a[j];
a[j]:=t;
end;
s:=0;
for i:=1 to m do
s:=s+a[i];
write(f1,abs(s));
close(f);close(f1);
end.
