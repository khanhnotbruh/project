Procedure ChenPhanTu( A:Mang20;Var N : Integer; k, X:Integer);
Var i :Integer;
Begin
For i:=N downto k+ 1 do
            A[i] := A[i-1];
      A[k] := X;
      N:=N+1;
End;
{ 2. Thu tuc xoa phantu }
Procedure XoaPhanTu( A:Mang20; Var N ,k:Integer);
Var i :Integer;
Begin
       For  i:=k to N-1 do
              A[i] := A[i+1];
       N:=N-1;
End;
